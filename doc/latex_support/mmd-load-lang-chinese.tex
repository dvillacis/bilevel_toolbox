\usepackage{xeCJK}
% [AutoFakeBold, AutoFakeSlant] can be used to fake bold and "italic"/slant automatically
\setCJKmainfont[BoldFont = * Bold, AutoFakeSlant]{Kaiti TC}
\setCJKsansfont[BoldFont = * Semibold, AutoFakeSlant]{PingFang TC} % for Mac. Windows: "Microsoft YaHei New"/"Microsoft Yahei". Alternatively, "Noto Sans CJK TC" looks almost the same as PingFang TC
\setCJKmonofont[BoldFont = * Bold, AutoFakeSlant]{Noto Sans Mono CJK TC} % Download Noto Sans CJK from https://www.google.com/get/noto/help/cjk/
